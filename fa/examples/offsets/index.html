from jalali_pandas.offsets import JalaliMonthEnd, JalaliWeek, FRIDAY

ts = JalaliTimestamp(1402, 6, 15)

next_month_end = ts + JalaliMonthEnd()
next_friday = ts + JalaliWeek(weekday=FRIDAY)
